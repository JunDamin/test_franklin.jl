# conduct bootstrapping with sample size n

def bootstrap(population, size, n):
    output = []
    for i in range(n):
        output.append(np.random.choice(population, size=size).mean())
    return output

fig, ax = plt.subplots(4, 2, figsize=(16, 24))
# 1000 resamples
for i, size in enumerate([10, 100, 1000, 10000]):
    output = bootstrap(samples[i], size, 1000)
    ax[i][0].hist(output, bins=50, range=(0, 60))
    ax[i][0].set_xlim(0, 60)
    ax[i][0].set_title(f"1000 means of size = {size}")
    ax[i][0].vlines(population.mean(), 0, 1000, color="red")
# 100000 resamples
